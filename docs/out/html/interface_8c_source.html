88 token_t get_tok() {
89 
90  int ch, finished = 0;
91  token_t tok = NONE_TOKEN;
92 
93  skip_ws();
94  init_char_buffer(scanner_buffer);
95 
96  while(!finished) {
97  ch = get_char();
98  switch(ch) {
99  case END_FILE:
100  file_flag++;
101  tok = END_OF_FILE;
102  finished++;
103  break;
104  case END_INPUT:
105  tok = END_OF_INPUT;
106  finished ++;
107  break;
108  case '\"': // read a double quoted string
109  tok = read_dquote();
110  finished ++;
111  break;
112  case '\'': // read a single quoted string
113  tok = read_squote();
114  finished ++;
115  break;
116  case '/': // may have a comment or an operator
117  tok = comment_or_operator();
118  skip_ws();
119  if(tok != NONE_TOKEN)
120  finished ++;
121  // continue if it was a comment
122  break;
123  default:
124  if(ch == END_FILE)
125  unget_char(ch);
126  else if(isdigit(ch)) { // beginning of a number. Could be hex, dec, or float
127  unget_char(ch);
128  tok = read_number_top();
129  if(tok != NONE_TOKEN)
130  finished ++;
131  }
132  else if(isalpha(ch)) { // could be a symbol or a keyword
133  unget_char(ch);
134  tok = read_word();
135  if(tok != NONE_TOKEN)
136  finished++;
137  }
138  else if(ispunct(ch)) { // some kind of operator (but not a '/' or a quote)
139  tok = read_punct(ch);
140  if(tok != NONE_TOKEN)
141  finished ++;
142  }
143  else {
144  warning("Unknown character, ignoring. (0x%02X) (%d) \'%c\'", ch, ch, ch);
145  }
146  }
147  }
148  return tok;
149 }
